144        Get DICOM tag values from a DICOM dataset.

150        tags : list[str] | list[list[int]]
151            List of DICOM tag addresses (keywords or hexadecimal values).
152
153        Returns
154        -------
155        dict[str, str]
156            Dictionary where keys are requested DICOM tags and values are

158        """
159        # Check if tags variable is a list of DICOM tags
160        if not isinstance(tags, list):
161            err_msg = f"Tags variable should be a list of tags, not a {type(tags)}."
162            logger.error(err_msg)
163            raise ValueError(err_msg)
164
165        # Loop over all DICOM tags
166        out_value = {}
167        for tag in tags:
168
169            # Check DICOM existence and get processed value
170            if tag in dataset:
171                out_value[tag] = dataset[tag].value
172
173            else:
174                out_value[tag] =  ""
175
176            # Format out_value
177            if dataset[tag].VR == "UI":
178                out_value[tag] = out_value[tag].replace(".","+")
179                out_value[tag] = sum(map(int, re.findall(r'[+-]?\d+', out_value[tag])))
180                out_value[tag] = str(hex(int(out_value[tag]))).upper()[2:]
181

371            0x00080022, 0x00080023, 0x00080050, 0x00100010,
372            0x00100020, 0x00100030, 0x00200010,
373        ]
374        values = [
375            new_station   , new_study_date, new_study_date  , new_study_date,

380
